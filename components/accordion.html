Accordion

Accordions display a list of items that can be individually expanded or collapsed to reveal associated content. Use them to progressively disclose information and reduce cognitive load on complex pages.

<div class="cue-accordion" role="list">

  <!-- Collapsed item -->
  <div class="cue-accordion__item" role="listitem">
    <button
      class="cue-accordion__trigger"
      aria-expanded="false"
      aria-controls="panel-1"
      id="trigger-1"
    >
      <span class="cue-accordion__label">Accordion Item</span>
      <span class="cue-accordion__icon material-icons" aria-hidden="true">add</span>
    </button>
    <div class="cue-accordion__panel" id="panel-1" role="region" aria-labelledby="trigger-1">
      <div class="cue-accordion__body">
        Content goes here.
      </div>
    </div>
  </div>

  <!-- Expanded item — add .cue-accordion__item--expanded -->
  <div class="cue-accordion__item cue-accordion__item--expanded" role="listitem">
    <button
      class="cue-accordion__trigger"
      aria-expanded="true"

/**
 * CUE Canvas — Accordion
 * Supports multiple items open simultaneously.
 */
function initAccordions() {
  const triggers = document.querySelectorAll('.cue-accordion__trigger');

  triggers.forEach(trigger => {

    trigger.addEventListener('click', () => {
      const item = trigger.closest('.cue-accordion__item');
      const isExpanded = item.classList.contains('cue-accordion__item--expanded');
      const icon = trigger.querySelector('.cue-accordion__icon');

      if (isExpanded) {
        item.classList.remove('cue-accordion__item--expanded');
        trigger.setAttribute('aria-expanded', 'false');
        icon.textContent = 'add';
      } else {
        item.classList.add('cue-accordion__item--expanded');
        trigger.setAttribute('aria-expanded', 'true');
        icon.textContent = 'remove';
      }
    });

    trigger.addEventListener('keydown', (e) => {
      const accordion = trigger.closest('.cue-accordion');
      const all = [...accordion.querySelectorAll('.cue-accordion__trigger')];
      const i = all.indexOf(trigger);

      if (e.key === 'ArrowDown') { e.preventDefault(); all[(i + 1) % all.length].focus(); }
      if (e.key === 'ArrowUp')   { e.preventDefault(); all[(i - 1 + all.length) % all.length].focus(); }
      if (e.key === 'Home')      { e.preventDefault(); all[0].focus(); }
      if (e.key === 'End')       { e.preventDefault(); all[all.length - 1].focus(); }
    });

  });
}

document.addEventListener('DOMContentLoaded', initAccordions);

Accessibility

Attribute / PatternElementNotes
aria-expanded .cue-accordion__trigger "true" when expanded, "false" when collapsed. Updated by JS on toggle.
aria-controls .cue-accordion__trigger References the ID of the associated panel element.
role="region" .cue-accordion__panel Identifies the panel as a landmark region for screen readers.
aria-labelledby .cue-accordion__panel References the trigger button ID so the panel is named by its trigger.
aria-hidden="true" .cue-accordion__icon Icon is decorative — state is communicated via aria-expanded.
:focus-visible .cue-accordion__trigger Focus ring shown on keyboard nav only, not on mouse click.
Arrow Up / Down .cue-accordion__trigger Moves focus between triggers within the accordion.
Home / End .cue-accordion__trigger Jumps focus to first or last trigger.
