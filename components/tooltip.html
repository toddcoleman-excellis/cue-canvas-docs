Tooltip

A small floating label that appears on hover or focus to provide supplementary context for an element. Used throughout CUE Back Office to clarify icon buttons, truncated labels, and form field hints. Four position variants — Top, Bottom, Left, Right.

Positions

Four positions. The caret arrow always points back toward the trigger. Hover each button to see the tooltip appear. The default position is Top.

<!-- Top (default) — CSS hover only, no JS needed -->
<div class="cue-tooltip-wrap">
  <button class="cue-btn cue-btn--neutral-stroke" type="button">
    Hover me
  </button>
  <div class="cue-tooltip" role="tooltip">
    <div class="cue-tooltip__arrow" aria-hidden="true"></div>
    Tooltip text
  </div>
</div>

<!-- Bottom -->

<!-- Disabled element pattern:
     Wrap disabled element in a <span> so hover still fires -->
<div class="cue-tooltip-wrap">
  <span style="display:inline-flex;">
    <button disabled>Save</button>
  </span>
  <div class="cue-tooltip" role="tooltip">
    <div class="cue-tooltip__arrow" aria-hidden="true"></div>
    Complete all required fields to save
  </div>
</div>
/* Wrapper — always position: relative */
.cue-tooltip-wrap { position: relative; display: inline-flex; align-items: center; justify-content: center; }

/* Panel — hidden by default */
.cue-tooltip {
  position: absolute; z-index: 700;
  background-color: var(--bg-default);
  border: 1px solid var(--border-default);   /* #575757 — darker than cards */
  border-radius: 8px; padding: 8px 12px;
  font-size: 0.75rem; font-weight: 700;
  color: var(--content-body-default);
  white-space: nowrap; pointer-events: none;
  opacity: 0; visibility: hidden;
  transition: opacity 0.15s ease, visibility 0.15s ease, transform 0.15s ease;
  /* Default: above trigger */
  bottom: calc(100% + 10px); left: 50%;
  transform: translateX(-50%) translateY(-2px);
}

/* Show on hover / focus-within */
.cue-tooltip-wrap:hover .cue-tooltip,
.cue-tooltip-wrap:focus-within .cue-tooltip {
  opacity: 1; visibility: visible;
  transform: translateX(-50%) translateY(0);
}

/* Arrow — CSS border trick */
.cue-tooltip__arrow {
  position: absolute; width: 0; height: 0; border-style: solid;
  /* Default (top): points down */
  top: 100%; left: 50%; transform: translateX(-50%);
  border-width: 7px 7px 0 7px;
  border-color: var(--border-default) transparent transparent transparent;
}
.cue-tooltip__arrow::after {
  content: ''; position: absolute;
  top: -8px; left: -6px;
  border-style: solid;
  border-width: 6px 6px 0 6px;
  border-color: var(--bg-default) transparent transparent transparent;
}

/* Position variants */
.cue-tooltip--bottom { bottom: auto; top: calc(100% + 10px); transform: translateX(-50%) translateY(2px); }
.cue-tooltip-wrap:hover .cue-tooltip--bottom { transform: translateX(-50%) translateY(0); }
.cue-tooltip--bottom .cue-tooltip__arrow { top: auto; bottom: 100%; border-width: 0 7px 7px; border-color: transparent transparent var(--border-default); }
.cue-tooltip--bottom .cue-tooltip__arrow::after { top: 2px; border-width: 0 6px 6px; border-color: transparent transparent var(--bg-default); }

.cue-tooltip--left { bottom: auto; top: 50%; left: auto; right: calc(100% + 10px); transform: translateY(-50%) translateX(-2px); }
.cue-tooltip-wrap:hover .cue-tooltip--left { transform: translateY(-50%) translateX(0); }
.cue-tooltip--left .cue-tooltip__arrow { top: 50%; left: 100%; transform: translateY(-50%); border-width: 7px 0 7px 7px; border-color: transparent transparent transparent var(--border-default); }
.cue-tooltip--left .cue-tooltip__arrow::after { top: -6px; left: -8px; border-width: 6px 0 6px 6px; border-color: transparent transparent transparent var(--bg-default); }

.cue-tooltip--right { bottom: auto; top: 50%; left: calc(100% + 10px); transform: translateY(-50%) translateX(2px); }
.cue-tooltip-wrap:hover .cue-tooltip--right { transform: translateY(-50%) translateX(0); }

// Tooltips work CSS-only (hover + focus-within) — no JS required
// for the standard hover pattern.

// Optional: Escape key dismissal for keyboard users
document.addEventListener('keydown', function(e) {
  if (e.key !== 'Escape') return;
  // Remove focus from any focused tooltip trigger
  if (document.activeElement) {
    document.activeElement.blur();
  }
});

// Optional: click-triggered tooltip (e.g. touch devices)
// Add data-tooltip-trigger="click" to .cue-tooltip-wrap
document.querySelectorAll('[data-tooltip-trigger="click"]').forEach(function(wrap) {
  wrap.addEventListener('click', function(e) {
    e.stopPropagation();
    var isOpen = wrap.classList.toggle('is-open');
    // Close others
    document.querySelectorAll('.cue-tooltip-wrap.is-open').forEach(function(other) {
      if (other !== wrap) other.classList.remove('is-open');
    });
  });
});

document.addEventListener('click', function() {
  document.querySelectorAll('.cue-tooltip-wrap.is-open').forEach(function(wrap) {
    wrap.classList.remove('is-open');
  });
});
