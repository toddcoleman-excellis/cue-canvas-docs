Step Process

A horizontal progress indicator for multi-step workflows. Each step has three states — Inactive, Current, and Complete. Used in wizards, checkout flows, onboarding sequences, and any multi-stage process where users need to track their position and progress.

Step states

Three states per step. The 4px indicator bar, label color, title weight, and optional check icon each change per state.

/**
 * setStep(stepsEl, currentIndex)
 *
 * Updates all .cue-step elements inside a .cue-steps container
 * to reflect the current step index (0-based).
 *
 * Steps before currentIndex → .is-complete
 * Steps at currentIndex     → .is-current
 * Steps after currentIndex  → (no modifier — inactive)
 */
function setStep(stepsEl, currentIndex) {
  const steps = [...stepsEl.querySelectorAll('.cue-step')];
  steps.forEach(function(step, i) {
    step.classList.remove('is-current', 'is-complete');
    if (i < currentIndex)       step.classList.add('is-complete');
    else if (i === currentIndex) step.classList.add('is-current');
  });
}

// Example: advance through a 4-step form
var currentStep = 0;
var totalSteps  = 4;
var stepsEl     = document.querySelector('.cue-steps');

function goToStep(index) {
  currentStep = Math.max(0, Math.min(index, totalSteps - 1));
  setStep(stepsEl, currentStep);
  // Show/hide content panels, update button states, etc.
}

document.getElementById('step-next').addEventListener('click', function() {
  goToStep(currentStep + 1);
});

document.getElementById('step-prev').addEventListener('click', function() {
  goToStep(currentStep - 1);
});
