Scheduler

The Resource Scheduler provides a horizontal timeline view for production line planning, personnel scheduling, and resource allocation. Resources are listed in a fixed left column with time slots scrolling horizontally. Activity blocks are placed in rows at the appropriate time positions.

Layout notes

ElementValueNotes
Cell width 160px per hour Fixed. Used as the base unit for all time calculations. Each 30-min half = 80px.
Row height 90px min-height Matches the Resource/Personnel block height. Can grow if content overflows.
Time header height 48px Sticky at top of scroll area. Corner cell in the fixed column matches this height.
Fixed column width 220px Contains resource name, status tag, and actions. Non-scrolling.
Activity block left (startHour - gridStartHour) × 160 + (startMin ÷ 60 × 160) Relative to the start of the scrollable grid. Grid start = 8:00 AM in this example.
Activity block width (durationMinutes ÷ 60) × 160 Minimum width should be ~80px (30 min) for legibility.
Activity block top timeHeaderHeight + (rowIndex × rowHeight) rowIndex is 0-based from the top of the grid rows container.
Activity block height rowHeight - 1px Subtract 1px to keep the row border visible.
Time indicator left Same formula as activity left Vertical amber line spanning full grid height. Circle marker sits at the top.
No card container — Unlike the Calendar, the Scheduler sits directly in the page content area without a card/module wrapper.

// Scheduler positioning.
// GRID_START_HOUR: first hour column (e.g. 8 = 8:00 AM).
// CELL_WIDTH: px width of each hour column (160px).
// ROW_HEIGHT:  px height of each resource row (90px).
// HEADER_HEIGHT: px height of the time header row (48px).

(function() {
  var GRID_START_HOUR = 8;
  var CELL_WIDTH      = 160;
  var ROW_HEIGHT      = 90;
  var HEADER_HEIGHT   = 48;

  function timeToLeft(hour, min) {
    return (hour - GRID_START_HOUR) * CELL_WIDTH + (min / 60 * CELL_WIDTH);
  }

  function rowTop(rowIndex) {
    return HEADER_HEIGHT + (rowIndex * ROW_HEIGHT);
  }

  function positionBlock(id, rowIndex, startH, startM, endH, endM) {
    var el = document.getElementById(id);
    if (!el) return;
    var durationMin = (endH - startH) * 60 + (endM - startM);
    el.style.left   = timeToLeft(startH, startM) + 'px';
    el.style.width  = (durationMin / 60 * CELL_WIDTH - 8) + 'px';
    el.style.top    = (rowTop(rowIndex) + 4) + 'px';
    el.style.height = (ROW_HEIGHT - 12) + 'px';
  }

  function layout() {
    // positionBlock(id, rowIndex, startH, startM, endH, endM)
    // rowIndex is 0-based (0 = first resource row)
    positionBlock('sa-1', 1, 9,  0,  11, 0 );
    positionBlock('sa-2', 3, 11, 0,  12, 30);
    positionBlock('sa-3', 4, 9,  0,  11, 0 );
    positionBlock('sa-4', 7, 8,  30, 11, 30);
    positionBlock('sa-5', 9, 9,  30, 11, 0 );

    // Current time indicator
    var ind = document.getElementById('sched-time-indicator');
    if (ind) ind.style.left = timeToLeft(10, 0) + 'px';
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', layout);
  } else {
    layout();
  }
  window.addEventListener('resize', layout);
})();
