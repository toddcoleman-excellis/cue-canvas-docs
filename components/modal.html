Modal

A dialog that appears over the page to focus user attention on a task or decision. The modal has three sections: a header with title and close button, a scrollable content body, and a footer with action buttons. Marked "Detach and Modify" in Figma — it is a template to be customized per use case.

Size variants

Two size variants. Default uses 24px padding and 18px button text. Small uses 12px/16px padding and 14px button text. Click to open each modal.

Anatomy

The modal is composed of three distinct zones. The header and footer are fixed height; the content body scrolls independently when content exceeds the max-height.

function openModal(modalId) {
  const overlay = document.getElementById(modalId + '-overlay');
  if (!overlay) return;
  overlay.classList.add('is-open');
  overlay.setAttribute('aria-hidden', 'false');
  document.body.style.overflow = 'hidden';

  // Focus the first focusable element inside the modal
  const modal = overlay.querySelector('.cue-modal');
  const firstFocusable = modal?.querySelector(
    'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
  );
  setTimeout(() => firstFocusable?.focus(), 50);
}

function closeModal(modalId) {
  const overlay = document.getElementById(modalId + '-overlay');
  if (!overlay) return;
  overlay.classList.remove('is-open');
  overlay.setAttribute('aria-hidden', 'true');
  document.body.style.overflow = '';

  // Return focus to the trigger that opened this modal
  const trigger = document.querySelector(`[data-opens="${modalId}"]`);
  trigger?.focus();
}

// Close when clicking the overlay backdrop (not the modal itself)
function handleOverlayClick(event, modalId) {
  if (event.target === event.currentTarget) {
    closeModal(modalId);
  }
}

// Close on Escape key
document.addEventListener('keydown', function(e) {
  if (e.key !== 'Escape') return;
  const open = document.querySelector('.cue-modal-overlay.is-open');
  if (open) {
    const modal = open.querySelector('.cue-modal');
    closeModal(modal?.id || '');
  }
});

// Focus trap — keep Tab inside the open modal
document.addEventListener('keydown', function(e) {
  if (e.key !== 'Tab') return;
  const overlay = document.querySelector('.cue-modal-overlay.is-open');
  if (!overlay) return;
  const modal = overlay.querySelector('.cue-modal');
  const focusable = modal.querySelectorAll(
    'button:not([disabled]), [href], input:not([disabled]), select:not([disabled]), textarea:not([disabled]), [tabindex]:not([tabindex="-1"])'
  );
  const first = focusable[0];
  const last  = focusable[focusable.length - 1];
  if (e.shiftKey && document.activeElement === first) {
    e.preventDefault(); last.focus();
  } else if (!e.shiftKey && document.activeElement === last) {
    e.preventDefault(); first.focus();
  }
});
