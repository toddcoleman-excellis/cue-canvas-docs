Side Navigation

The primary left-side navigation panel for CUE Back Office. Supports section labels, collapsible two-level hierarchies, and icon-labelled items. The active item is highlighted with the primary brand blue and a 3px amber left border. Follows the ShadCN sidebar-02 interaction pattern.

Item states

Four states apply to both top-level items and child items. The active state always shows the 3px amber left border.

// Toggle a top-level collapsible item open/closed
function toggleNav(itemId) {
  const item = document.getElementById(itemId);
  if (!item) return;
  item.classList.toggle('is-open');
}

// Activate a child or sub-child item (deactivates all others)
function activateNav(btn, pageTitle) {
  // Clear all active states
  document.querySelectorAll('.cue-sidenav__child.is-active, .cue-sidenav__subchild.is-active')
    .forEach(el => el.classList.remove('is-active'));

  btn.classList.add('is-active');

  // Optional: update page title / breadcrumb
  const titleEl = document.getElementById('demo-page-title');
  if (titleEl) titleEl.textContent = btn.textContent.trim();
}

// Activate a top-level leaf item (no children)
function activateNavItem(btn, pageTitle) {
  document.querySelectorAll('.cue-sidenav__item-btn.is-active')
    .forEach(el => el.classList.remove('is-active'));
  document.querySelectorAll('.cue-sidenav__child.is-active, .cue-sidenav__subchild.is-active')
    .forEach(el => el.classList.remove('is-active'));

  btn.classList.add('is-active');

  const titleEl = document.getElementById('demo-page-title');
  if (titleEl) titleEl.textContent = pageTitle;
}

// Toggle a child item that itself has sub-children (level 3)
function toggleSubNav(childBtn) {
  childBtn.classList.toggle('is-open');
  // The adjacent .cue-sidenav__subchildren opens via CSS sibling selector
}
