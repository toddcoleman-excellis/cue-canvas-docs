Icon Reference

Dropdown Menu

A contextual menu panel triggered by a button. Used throughout CUE Back Office for actions on records, row-level context menus, and "Add New" create flows. The panel appears anchored below the trigger button and can contain labeled actions, icons, submenu indicators, and a divider for visual grouping.

Actions / Context Menu

Used for record-level actions. Triggered by an "Actions" label button or a more_horiz icon button. Contains labelled actions with leading icons. A divider separates destructive actions (Delete) from the rest. Download has a trailing chevron_right indicating a submenu.

Neutral — stroke trigger

Icon trigger — more_horiz

Panel — standalone view

Add / Create Menu

Used for "Add New" or "Create" flows. Typically triggered by a primary button. Items represent entity types to create (Contact, Folder, Row). Trailing chevron_right on items indicates a further step or sub-flow.

Primary trigger

Panel — standalone view

function toggleDropdown(id, force) {
  const el = document.getElementById(id);
  if (!el) return;
  const isOpen = force !== undefined ? force : !el.classList.contains('is-open');

  // Close all other open dropdowns first
  document.querySelectorAll('.cue-dropdown.is-open').forEach(d => {
    if (d.id !== id) {
      d.classList.remove('is-open');
      d.querySelector('[aria-expanded]')?.setAttribute('aria-expanded', 'false');
    }
  });

  el.classList.toggle('is-open', isOpen);
  el.querySelector('[aria-expanded]')?.setAttribute('aria-expanded', isOpen);

  // Focus first item when opening
  if (isOpen) {
    setTimeout(() => el.querySelector('.cue-dropdown__item')?.focus(), 50);
  }
}

// Keyboard navigation within an open dropdown
document.addEventListener('keydown', function(e) {
  const open = document.querySelector('.cue-dropdown.is-open');
  if (!open) return;

  const items = [...open.querySelectorAll('.cue-dropdown__item')];
  const focused = document.activeElement;
  const idx = items.indexOf(focused);

  if (e.key === 'ArrowDown') {
    e.preventDefault();
    items[(idx + 1) % items.length]?.focus();
  } else if (e.key === 'ArrowUp') {
    e.preventDefault();
    items[(idx - 1 + items.length) % items.length]?.focus();
  } else if (e.key === 'Escape') {
    const trigger = open.querySelector('[aria-expanded]');
    toggleDropdown(open.id, false);
    trigger?.focus();
  } else if (e.key === 'Tab') {
    toggleDropdown(open.id, false);
  }
});

// Close on outside click
document.addEventListener('click', function(e) {
  document.querySelectorAll('.cue-dropdown.is-open').forEach(el => {
    if (!el.contains(e.target)) {
      el.classList.remove('is-open');
      el.querySelector('[aria-expanded]')?.setAttribute('aria-expanded', 'false');
    }
  });
});
