Popover

A small floating panel anchored to a trigger element. Used for contextual information, quick actions, or supplementary content without leaving the current view. Triggered by click or hover. Dismisses on outside click or Escape. Content is fully flexible.

Basic

The base popover — a plain text message anchored below the trigger. Click the button to toggle.

Simple message

Simple popover message on hover or click.

Icon trigger

This field requires a unique identifier.

Above trigger

Popover positioned above the trigger.

Rich content

The popover content area accepts any HTML. Common patterns include a title + body description, a list of quick actions, or form elements.

// Toggle a popover open/closed
function togglePopover(id, force) {
  const wrap = document.getElementById(id);
  if (!wrap) return;
  const isOpen = force !== undefined ? force : !wrap.classList.contains('is-open');

  // Close all other open popovers first
  document.querySelectorAll('.cue-popover-wrap.is-open').forEach(w => {
    if (w.id !== id) w.classList.remove('is-open');
  });

  wrap.classList.toggle('is-open', isOpen);
}

// Close on outside click
document.addEventListener('click', function(e) {
  document.querySelectorAll('.cue-popover-wrap.is-open').forEach(wrap => {
    if (!wrap.contains(e.target)) {
      wrap.classList.remove('is-open');
    }
  });
});

// Close on Escape
document.addEventListener('keydown', function(e) {
  if (e.key !== 'Escape') return;
  document.querySelectorAll('.cue-popover-wrap.is-open').forEach(wrap => {
    wrap.classList.remove('is-open');
  });
});

// Hover trigger (optional — use instead of or alongside click)
function initHoverPopover(id) {
  const wrap = document.getElementById(id);
  if (!wrap) return;
  wrap.addEventListener('mouseenter', () => togglePopover(id, true));
  wrap.addEventListener('mouseleave', () => togglePopover(id, false));
}
