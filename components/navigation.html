Navigation

Four navigation patterns used throughout CUE Back Office. Tab Navigation for horizontal primary switching. Side Navigation for vertical secondary menus. Pill Navigation for filter-style category selection. Subnav Tabs for secondary page navigation with overflow scroll support.

Tab Navigation

Horizontal tabs with a 2px amber bottom indicator on the active item. Supports optional leading icons. Active item is Bold; default items are Regular. Click tabs to switch.

With icons

Text only

Side Navigation

Vertical navigation with a 3px amber left border on the active item. Two variants: with icons (12px gap) and text only (16px gap). Click items to switch.

With icons

Text only

Pill Navigation

Horizontal pill-shaped items on a white container. Active item gets a filled accent-blue background (#61c5ff). Used for filter-style or category navigation. Click items to switch.

With icons

Text only

Subnav Tabs

Secondary navigation bar used throughout CUE Back Office pages. Items use 14px SemiBold text in uppercase. The active item is Bold with a 4px amber bottom bar. When items overflow the container width, the bar scrolls horizontally with a gradient fade and chevron arrow. A cart count badge variant is also available.

// Tab Navigation — activate clicked item, deactivate siblings
function activateTab(btn) {
  const nav = btn.closest('.cue-tab-nav');
  nav.querySelectorAll('.cue-tab-nav__item').forEach(b => b.classList.remove('is-active'));
  btn.classList.add('is-active');
}

// Side Navigation — activate clicked item, deactivate siblings
function activateSideNav(btn) {
  const nav = btn.closest('.cue-side-nav');
  nav.querySelectorAll('.cue-side-nav__item').forEach(b => b.classList.remove('is-active'));
  btn.classList.add('is-active');
}

// Pill Navigation — activate clicked item, deactivate siblings
function activatePillNav(btn) {
  const nav = btn.closest('.cue-pill-nav');
  nav.querySelectorAll('.cue-pill-nav__item').forEach(b => b.classList.remove('is-active'));
  btn.classList.add('is-active');
}

// Subnav Tabs — activate clicked tab
function activateSubnav(btn) {
  const nav = btn.closest('.cue-subnav');
  nav.querySelectorAll('.cue-subnav__tab').forEach(b => b.classList.remove('is-active'));
  btn.classList.add('is-active');
}

// Subnav overflow scroll — delta: 1 = right, -1 = left
function scrollSubnav(navId, delta) {
  const nav = document.getElementById(navId);
  const track = nav.querySelector('.cue-subnav__track');
  if (!track) return;
  track.scrollBy({ left: delta * 200, behavior: 'smooth' });
  setTimeout(() => updateSubnavArrows(navId), 300);
}

// Show/hide arrows based on scroll position
function updateSubnavArrows(navId) {
  const nav   = document.getElementById(navId);
  const track = nav.querySelector('.cue-subnav__track');
  const prev  = nav.querySelector('.cue-subnav__arrow:first-of-type, #' + navId + '-prev');
  const next  = nav.querySelector('.cue-subnav__arrow:last-of-type,  #' + navId + '-next');
  const gradL = document.getElementById(navId + '-grad-left');
  const gradR = document.getElementById(navId + '-grad-right');
  if (!track) return;

  const atStart = track.scrollLeft <= 0;
  const atEnd   = track.scrollLeft + track.clientWidth >= track.scrollWidth - 2;

  prev?.classList.toggle('is-hidden', atStart);
  gradL?.classList.toggle('is-hidden', atStart);
  next?.classList.toggle('is-hidden', atEnd);
  gradR?.classList.toggle('is-hidden', atEnd);
}

// Initialise overflow subnavs on load
document.addEventListener('DOMContentLoaded', function() {
  document.querySelectorAll('.cue-subnav').forEach(function(nav) {
    const track = nav.querySelector('.cue-subnav__track');
    if (!track) return;
    track.addEventListener('scroll', function() {
      updateSubnavArrows(nav.id);
    });
    updateSubnavArrows(nav.id);
  });
});
