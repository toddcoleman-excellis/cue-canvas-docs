Date Picker

A form field that opens a calendar panel for date selection. Available in two variants: Single Date for selecting one date, and Date Range for selecting a start and end date with built-in preset shortcuts (Today, Last 30 Days, etc.).

Single Date Picker

Click the input to open the calendar. Navigate months with the arrow buttons. Click a date to select it — the panel closes and the input shows the selected date in MM/DD/YYYY format.

Date Range Picker

Click the input to open the range panel. Preset shortcuts on the left apply a date range immediately. Custom Range allows clicking a start and end date across the two-month calendar view. Clear resets the selection; Apply confirms and closes the panel.

/* ── SHARED HELPERS ───────────────────────────────────────────── */

const MONTH_NAMES = ['January','February','March','April','May','June',
  'July','August','September','October','November','December'];
const DOW = ['S','M','T','W','T','F','S'];

// State store: { viewYear, viewMonth, selected: Date|null }
const dpState = {};

function initDP(id, selectedDate) {
  const now = selectedDate || new Date();
  dpState[id] = {
    viewYear: now.getFullYear(),
    viewMonth: now.getMonth(),
    selected: selectedDate || null,
  };
  renderSingleCalendar(id);
}

function renderSingleCalendar(id) {
  const s = dpState[id];
  const grid = document.getElementById(id + '-grid');
  const label = document.getElementById(id + '-label');
  if (!grid || !label) return;

  label.textContent = MONTH_NAMES[s.viewMonth] + ' ' + s.viewYear;
  grid.innerHTML = '';

  // Day-of-week headers
  DOW.forEach(d => {
    const cell = document.createElement('div');
    cell.className = 'cue-dp-day-cell is-dow';
    cell.textContent = d;
    grid.appendChild(cell);
  });

  const firstDay = new Date(s.viewYear, s.viewMonth, 1).getDay();
  const daysInMonth = new Date(s.viewYear, s.viewMonth + 1, 0).getDate();
  const daysInPrev = new Date(s.viewYear, s.viewMonth, 0).getDate();
  const today = new Date(); today.setHours(0,0,0,0);

  // Prev month overflow
  for (let i = firstDay - 1; i >= 0; i--) {
    const cell = document.createElement('div');
    cell.className = 'cue-dp-day-cell is-other-month';
    cell.textContent = daysInPrev - i;
    grid.appendChild(cell);
  }

  // Current month days
  for (let d = 1; d <= daysInMonth; d++) {
    const date = new Date(s.viewYear, s.viewMonth, d);
    date.setHours(0,0,0,0);
    const cell = document.createElement('button');
    cell.type = 'button';
    cell.className = 'cue-dp-day-cell';
    cell.textContent = d;
    cell.setAttribute('aria-label', date.toDateString());
    if (date.getTime() === today.getTime()) cell.classList.add('is-today');
    if (s.selected && date.getTime() === s.selected.getTime()) cell.classList.add('is-selected');
    cell.addEventListener('click', function(e) {
      e.stopPropagation();
      selectDate(id, date);
    });
    grid.appendChild(cell);
  }

  // Next month overflow
  const totalCells = firstDay + daysInMonth;
  const remaining = totalCells % 7 === 0 ? 0 : 7 - (totalCells % 7);
  for (let i = 1; i <= remaining; i++) {
    const cell = document.createElement('div');
    cell.className = 'cue-dp-day-cell is-other-month';
    cell.textContent = i;
    grid.appendChild(cell);
  }
}

function selectDate(id, date) {
  dpState[id].selected = date;
  const display = document.getElementById(id + '-display');
  if (display) {
    const m = date.getMonth() + 1;
    const d = date.getDate();
    const y = date.getFullYear();
    display.textContent = `${String(m).padStart(2,'0')}/${String(d).padStart(2,'0')}/${y}`;
    display.closest('.cue-dp-input').classList.add('has-value');
  }
  renderSingleCalendar(id);
  toggleDatepicker(id, false);
}

function toggleDatepicker(id, force) {
  const wrap = document.getElementById(id);
  if (!wrap) return;
  const isOpen = force !== undefined ? force : !wrap.classList.contains('is-open');
  wrap.classList.toggle('is-open', isOpen);
  wrap.querySelector('.cue-dp-input')?.setAttribute('aria-expanded', isOpen);
  if (isOpen && !dpState[id]) initDP(id, null);
  if (isOpen) renderSingleCalendar(id);
}

function navMonth(id, delta) {
  const s = dpState[id];
  if (!s) return;
  s.viewMonth += delta;
  if (s.viewMonth > 11) { s.viewMonth = 0; s.viewYear++; }
  if (s.viewMonth < 0)  { s.viewMonth = 11; s.viewYear--; }
  renderSingleCalendar(id);
}


/* ── DATE RANGE ───────────────────────────────────────────────── */

const drState = {};

function initDR(id, startDate, endDate) {
  const now = new Date();
  drState[id] = {
    viewYear: now.getFullYear(),
    viewMonth: now.getMonth() - 1, // show prev+current by default
    rangeStart: startDate || null,
    rangeEnd:   endDate || null,
    picking:    0, // 0=waiting for start, 1=waiting for end
    activePreset: null,
  };
  renderRangeCalendars(id);
}

function renderRangeCalendars(id) {
  const s = drState[id];
  if (!s) return;

  // Left = s.viewMonth, Right = s.viewMonth+1
  renderRangeMonth(id, 'left',  s.viewYear, s.viewMonth);
  const rYear = s.viewMonth === 11 ? s.viewYear + 1 : s.viewYear;
  const rMonth = s.viewMonth === 11 ? 0 : s.viewMonth + 1;
  renderRangeMonth(id, 'right', rYear, rMonth);
}

function renderRangeMonth(id, side, year, month) {
  const s = drState[id];
  const grid  = document.getElementById(id + '-grid-' + side);
  const label = document.getElementById(id + '-label-' + side);
  if (!grid || !label) return;

  label.textContent = MONTH_NAMES[month] + ' ' + year;
  grid.innerHTML = '';

  DOW.forEach(d => {
    const cell = document.createElement('div');
    cell.className = 'cue-dp-day-cell is-dow';
    cell.textContent = d;
    grid.appendChild(cell);
  });

  const firstDay = new Date(year, month, 1).getDay();
  const daysInMonth = new Date(year, month + 1, 0).getDate();
  const daysInPrev  = new Date(year, month, 0).getDate();
  const today = new Date(); today.setHours(0,0,0,0);

  for (let i = firstDay - 1; i >= 0; i--) {
    const cell = document.createElement('div');
    cell.className = 'cue-dp-day-cell is-other-month';
    cell.textContent = daysInPrev - i;
    grid.appendChild(cell);
  }

  for (let d = 1; d <= daysInMonth; d++) {
    const date = new Date(year, month, d);
    date.setHours(0,0,0,0);
    const cell = document.createElement('button');
    cell.type = 'button';
    cell.className = 'cue-dp-day-cell';
    cell.textContent = d;

    const t = date.getTime();
    const rs = s.rangeStart ? s.rangeStart.getTime() : null;
    const re = s.rangeEnd   ? s.rangeEnd.getTime()   : null;

    if (date.getTime() === today.getTime()) cell.classList.add('is-today');
    if (rs && t === rs && re && t === re) cell.classList.add('is-selected');
    else if (rs && t === rs) cell.classList.add('is-range-start');
    else if (re && t === re) cell.classList.add('is-range-end');
    else if (rs && re && t > rs && t < re) cell.classList.add('is-in-range');

    cell.addEventListener('click', function(e) {
      e.stopPropagation();
      pickRangeDate(id, date);
    });
    grid.appendChild(cell);
  }

  const totalCells = firstDay + daysInMonth;
  const remaining = totalCells % 7 === 0 ? 0 : 7 - (totalCells % 7);
  for (let i = 1; i <= remaining; i++) {
    const cell = document.createElement('div');
    cell.className = 'cue-dp-day-cell is-other-month';
    cell.textContent = i;
    grid.appendChild(cell);
  }
}

function pickRangeDate(id, date) {
  const s = drState[id];
  if (s.picking === 0 || (s.rangeStart && s.rangeEnd)) {
    // Start new selection
    s.rangeStart = date;
    s.rangeEnd   = null;
    s.picking    = 1;
    setPresetActive(id, null);
  } else {
    // Complete the range
    if (date < s.rangeStart) {
      s.rangeEnd   = s.rangeStart;
      s.rangeStart = date;
    } else {
      s.rangeEnd = date;
    }
    s.picking = 0;
  }
  renderRangeCalendars(id);
}

function applyDaterange(id) {
  const s = drState[id];
  const display = document.getElementById(id + '-display');
  if (display && s.rangeStart) {
    const fmt = d => `${d.getMonth()+1}/${d.getDate()}/${String(d.getFullYear()).slice(2)}`;
    display.textContent = s.rangeEnd
      ? `${fmt(s.rangeStart)} - ${fmt(s.rangeEnd)}`
      : fmt(s.rangeStart);
    display.closest('.cue-dp-input').classList.add('has-value');
  }
  toggleDaterange(id, false);
}

function clearDaterange(id) {
  const s = drState[id];
  s.rangeStart = null;
  s.rangeEnd   = null;
  s.picking    = 0;
  s.activePreset = null;
  const display = document.getElementById(id + '-display');
  if (display) {
    display.textContent = 'Date Range';
    display.closest('.cue-dp-input').classList.remove('has-value');
  }
  setPresetActive(id, null);
  renderRangeCalendars(id);
}

function applyPreset(id, preset) {
  const s = drState[id];
  const today = new Date(); today.setHours(0,0,0,0);
  let start, end = new Date(today);

  if (preset === 'today') {
    start = new Date(today);
  } else if (preset === 'last30') {
    start = new Date(today); start.setDate(start.getDate() - 30);
  } else if (preset === 'last90') {
    start = new Date(today); start.setDate(start.getDate() - 90);
  } else if (preset === 'thisYear') {
    start = new Date(today.getFullYear(), 0, 1);
    end   = new Date(today.getFullYear(), 11, 31);
  } else if (preset === 'lastYear') {
    start = new Date(today.getFullYear()-1, 0, 1);
    end   = new Date(today.getFullYear()-1, 11, 31);
  } else if (preset === 'custom') {
    s.rangeStart = null; s.rangeEnd = null; s.picking = 0;
    setPresetActive(id, 'custom');
    renderRangeCalendars(id);
    return;
  }

  s.rangeStart = start;
  s.rangeEnd   = end;
  s.picking    = 0;
  // Show the start month on the left calendar
  s.viewYear   = start.getFullYear();
  s.viewMonth  = start.getMonth();
  setPresetActive(id, preset);
  renderRangeCalendars(id);
}

function setPresetActive(id, preset) {
  const panel = document.getElementById(id + '-panel');
  if (!panel) return;
  panel.querySelectorAll('.cue-dr-preset-btn').forEach(btn => {
    btn.classList.remove('is-active');
  });
  if (preset) {
    const map = { today:'Today', last30:'Last 30 Days', last90:'Last 90 Days',
                  thisYear:'This Year', lastYear:'Last Year', custom:'Custom Range' };
    panel.querySelectorAll('.cue-dr-preset-btn').forEach(btn => {
      if (btn.textContent.trim() === map[preset]) btn.classList.add('is-active');
    });
  }
}

function toggleDaterange(id, force) {
  const wrap = document.getElementById(id);
  if (!wrap) return;
  const isOpen = force !== undefined ? force : !wrap.classList.contains('is-open');
  wrap.classList.toggle('is-open', isOpen);
  wrap.querySelector('.cue-dp-input')?.setAttribute('aria-expanded', isOpen);
  if (isOpen && !drState[id]) initDR(id, null, null);
  if (isOpen) renderRangeCalendars(id);
}

function navRangeMonth(id, delta) {
  const s = drState[id];
  if (!s) return;
  s.viewMonth += delta;
  if (s.viewMonth > 11) { s.viewMonth = 0; s.viewYear++; }
  if (s.viewMonth < 0)  { s.viewMonth = 11; s.viewYear--; }
  renderRangeCalendars(id);
}

/* Close on outside click / Escape */

    }
  });
});

  document.querySelectorAll('.cue-dp-wrap.is-open').forEach(wrap => {
    wrap.classList.remove('is-open');
    wrap.querySelector('[aria-expanded]')?.setAttribute('aria-expanded','false');
  });
