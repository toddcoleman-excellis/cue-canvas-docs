Tags

Compact pill-shaped labels — also called chips — used to represent filters, active selections, categories, or metadata values. Two variants: Filled (primary brand) for prominent selections and Outlined (neutral) for lighter labeling. Both support an optional dismiss button for removable tags.

Filled

Primary brand background with white text. Used for active filter selections or prominent category tags. Click the × to remove a tag.

Input + tag pattern

A common usage pattern — type a value and press Enter or comma to add it as a tag. Used in CUE Back Office for assigning categories, tags, or multi-value fields.

// Remove a tag when its × button is clicked
function removeTag(removeBtn) {
  removeBtn.closest('.cue-tag').remove();
}

// Toggle selected state on an outlined tag
function toggleTagSelected(tag) {
  tag.classList.toggle('is-selected');
}

// Input + tag pattern: add tag on Enter or comma
function handleTagInput(e) {
  const input = e.target;
  const value = input.value.trim().replace(/,$/, '').trim();

  if ((e.key === 'Enter' || e.key === ',') && value) {
    e.preventDefault();
    addInputTag(value);
    input.value = '';
  }

  // Remove last tag on Backspace if input is empty
  if (e.key === 'Backspace' && !input.value) {
    const wrap  = input.closest('[id$="-wrap"], .tag-input-wrap');
    const tags  = wrap.querySelectorAll('.cue-tag');
    const last  = tags[tags.length - 1];
    if (last) last.remove();
  }
}

function addInputTag(value) {
  const input = document.getElementById('tag-input');
  const wrap  = input.parentElement;
  const tag   = document.createElement('span');
  tag.className = 'cue-tag cue-tag--filled cue-tag--sm';
  tag.innerHTML = `
    <span class="cue-tag__label">${value}</span>
    <button class="cue-tag__remove" type="button" aria-label="Remove ${value}"
      onclick="this.closest('.cue-tag').remove()">
      <span class="material-icons" aria-hidden="true">close</span>
    </button>`;
  wrap.insertBefore(tag, input);
}
