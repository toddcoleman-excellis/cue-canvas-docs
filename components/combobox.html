Combobox & Multiselect

Two enhanced select field components built on top of the base Form Field token set. The Combobox provides single-selection with an autocomplete search filter. The Multiselect provides multiple-selection with checkboxes and a search field inside the dropdown panel.

Combobox — Single Select with Search

A select field with an inline search input that filters options as the user types. When closed the trigger shows the selected value or placeholder text. When opened the trigger becomes a search input and the dropdown shows filtered results. Selecting an option closes the panel and populates the field.

/* ── COMBOBOX ─────────────────────────────────────────────────── */

function toggleCombobox(id) {
  const el = document.getElementById(id);
  const isOpen = el.classList.toggle('is-open');
  el.querySelector('.cue-combobox__trigger')
    .setAttribute('aria-expanded', isOpen);
  if (isOpen) {
    el.querySelector('.cue-combobox__panel-search input')?.focus();
  }
}

function selectCombobox(id, option) {
  const el = document.getElementById(id);
  const value = option.dataset.value;
  // Clear previous selection
  el.querySelectorAll('.cue-combobox__option').forEach(o => {
    o.classList.remove('is-selected');
    o.setAttribute('aria-selected', 'false');
  });
  // Mark new selection
  option.classList.add('is-selected');
  option.setAttribute('aria-selected', 'true');
  // Update display
  const display = document.getElementById(id + '-display');
  if (display) {
    display.textContent = value;
    display.className = 'cue-combobox__selected-text';
  }
  // Close panel
  el.classList.remove('is-open');
  el.querySelector('.cue-combobox__trigger').setAttribute('aria-expanded', 'false');
}

function clearCombobox(id, event) {
  event.stopPropagation();
  const el = document.getElementById(id);
  el.querySelectorAll('.cue-combobox__option').forEach(o => {
    o.classList.remove('is-selected');
    o.setAttribute('aria-selected', 'false');
  });
  const display = document.getElementById(id + '-display');
  if (display) {
    display.textContent = 'Select';
    display.className = 'cue-combobox__placeholder';
  }
}

function filterCombobox(id, query) {
  const el = document.getElementById(id);
  const q = query.toLowerCase().trim();
  el.querySelectorAll('.cue-combobox__option').forEach(opt => {
    const text = opt.dataset.value?.toLowerCase() || '';
    opt.classList.toggle('is-hidden', q !== '' && !text.includes(q));
  });
}


/* ── MULTISELECT ──────────────────────────────────────────────── */

function toggleMultiselect(id) {
  const el = document.getElementById(id);
  const isOpen = el.classList.toggle('is-open');
  el.querySelector('.cue-multiselect__trigger')
    .setAttribute('aria-expanded', isOpen);
  if (isOpen) {
    el.querySelector('.cue-multiselect__search input')?.focus();
  }
}

function toggleMultiselectOption(id, option) {
  option.classList.toggle('is-selected');
  option.setAttribute('aria-selected',
    option.classList.contains('is-selected'));
  syncAllCheckbox(id);
  updateMultiselectTags(id);
}

function toggleAll(id, allOption) {
  const el = document.getElementById(id);
  const willSelect = !allOption.classList.contains('is-selected');
  el.querySelectorAll('.cue-multiselect__option').forEach(o => {
    o.classList.toggle('is-selected', willSelect);
    o.setAttribute('aria-selected', willSelect);
  });
  updateMultiselectTags(id);
}

function syncAllCheckbox(id) {
  const el = document.getElementById(id);
  const all = el.querySelector('.cue-multiselect__option.is-all');
  if (!all) return;
  const options = el.querySelectorAll('.cue-multiselect__option:not(.is-all)');
  const allSelected = [...options].every(o => o.classList.contains('is-selected'));
  all.classList.toggle('is-selected', allSelected);
}

function updateMultiselectTags(id) {
  const el = document.getElementById(id);
  const tagsEl = document.getElementById(id + '-tags');
  if (!tagsEl) return;
  const MAX_TAGS = 3;
  const selected = [...el.querySelectorAll(
    '.cue-multiselect__option:not(.is-all).is-selected'
  )].map(o => o.dataset.value).filter(Boolean);
  if (selected.length === 0) {
    tagsEl.innerHTML = '<span class="cue-multiselect__placeholder">Select</span>';
  } else {
    const shown = selected.slice(0, MAX_TAGS);
    const overflow = selected.length - MAX_TAGS;
    tagsEl.innerHTML = shown.map(v =>
      `<span class="cue-multiselect__tag">${v}</span>`
    ).join('') + (overflow > 0
      ? `<span class="cue-multiselect__overflow">+${overflow}</span>`
      : '');
  }
}

function filterMultiselect(id, query) {
  const el = document.getElementById(id);
  const q = query.toLowerCase().trim();
  el.querySelectorAll('.cue-multiselect__option:not(.is-all)').forEach(opt => {
    const text = opt.dataset.value?.toLowerCase() || '';
    opt.classList.toggle('is-hidden', q !== '' && !text.includes(q));
  });
}

/* Close on outside click */
document.addEventListener('click', function(e) {
  document.querySelectorAll('.cue-combobox.is-open, .cue-multiselect.is-open')
    .forEach(el => {
      if (!el.contains(e.target)) {
        el.classList.remove('is-open');
        el.querySelector('[aria-expanded]')?.setAttribute('aria-expanded', 'false');
      }
    });
});

/* Close on Escape */
document.addEventListener('keydown', function(e) {
  if (e.key === 'Escape') {
    document.querySelectorAll('.cue-combobox.is-open, .cue-multiselect.is-open')
      .forEach(el => {
        el.classList.remove('is-open');
        const trigger = el.querySelector('[aria-expanded]');
        trigger?.setAttribute('aria-expanded', 'false');
        trigger?.focus();
      });
  }
});
